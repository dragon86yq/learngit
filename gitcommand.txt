// 自报家门
//--global 参数，表示这台机器上的所有git仓库都是用这个配置
git config --global user.name ""
git config --global user.email ""

//初始化
git init

//把文件添加到仓库
git add readme.txt

//把文件提交到仓库
git commit -m "xxx"

//状态查看
git status
git diff

//查看提交信息
git log

//回退到上一个版本
git reset --hard HEAD^
//查看
cat readme.txt

//查看历史
git reflog

//撤销操作
git checkout -- readme.txt

//删除
rm test.txt

//删除并提交
git rm test.txt
git commit -m "remove test.txt"

//远程仓库
//创建SSH Key。输入下面命令后，一路回车。产生id_rsa和id_rsa.pub，这两个文件目录(/c/Users/slyq/.ssh/id_rsa)
ssh-keygen -t rsa -C "dragon86yq@gmail.com"

//将本地仓库推送到GitHub仓库
git remote add origin git@github.com:dragon86yq/learngit.git

//本地库所有内容推送到远程库上
//-u Git不但会把本地的master分支内容推送的远程新的master分支，还会把本地的master分支和远程的master分支关联起来，在以后的推送或者拉取时就可以简化命令。
git push -u origin master
